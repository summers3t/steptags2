        feather.replace();

        function addTeamMember() {
            const email = document.getElementById('inviteEmail').value;
            const role = document.getElementById('memberRole').value;
            
            if (!email) return;
            
            // Generate a random avatar based on email
            const avatarSeed = email.split('').reduce((acc, char) => acc + char.charCodeAt(0), 0);
            const avatarUrl = `http://static.photos/people/200x200/${avatarSeed}`;
            
            const memberDiv = document.createElement('div');
            memberDiv.className = 'relative inline-block';
            memberDiv.innerHTML = `
                <img src="${avatarUrl}" alt="${email}" class="w-8 h-8 rounded-full border-2 border-white">
                <span class="absolute -bottom-1 -right-1 bg-indigo-100 text-indigo-800 text-xs px-1 rounded-full">${role.charAt(0).toUpperCase()}</span>
            `;
            
            document.getElementById('teamMembersList').appendChild(memberDiv);
            
            // Clear input
            document.getElementById('inviteEmail').value = '';
            
            // Show success message
            const toast = document.createElement('div');
            toast.className = 'fixed bottom-4 right-4 px-4 py-2 bg-green-100 text-green-800 rounded-md shadow-md';
            toast.textContent = `${email} added as ${role}`;
            document.body.appendChild(toast);
            
            setTimeout(() => {
                toast.remove();
            }, 3000);
        }

        function openInviteModal() {
            document.getElementById('inviteModal').classList.remove('hidden');
        }

        function closeInviteModal() {
            document.getElementById('inviteModal').classList.add('hidden');
        }

        function sendInvites() {
            const emails = document.getElementById('inviteEmails').value;
            const role = document.getElementById('bulkRole').value;
            
            if (!emails) {
                alert('Please enter at least one email address');
                return;
            }
            
            const emailList = emails.split(',').map(email => email.trim());
            
            emailList.forEach(email => {
                if (email) {
                    // In a real app, you would send an invitation email here
                    console.log(`Invite sent to ${email} as ${role}`);
                }
            });
            
            // Show success message
            const toast = document.createElement('div');
            toast.className = 'fixed bottom-4 right-4 px-4 py-2 bg-green-100 text-green-800 rounded-md shadow-md';
            toast.textContent = `Invites sent to ${emailList.length} members`;
            document.body.appendChild(toast);
            
            setTimeout(() => {
                toast.remove();
            }, 3000);
            
            closeInviteModal();
        }
