    <script>
        // icons
        document.addEventListener('DOMContentLoaded', () => { window.feather && window.feather.replace(); });

        // simple password toggle (no layout shift)
        const pass = document.getElementById('password');
        const toggle = document.getElementById('toggle-pass');
        toggle?.addEventListener('click', () => {
            const visible = pass.getAttribute('type') === 'text';
            pass.setAttribute('type', visible ? 'password' : 'text');
            toggle.setAttribute('aria-label', visible ? 'Show password' : 'Hide password');
            // swap icon
            toggle.innerHTML = '';
            const i = document.createElement('i');
            i.setAttribute('data-feather', visible ? 'eye' : 'eye-off');
            i.className = 'w-5 h-5';
            toggle.appendChild(i);
            window.feather && window.feather.replace();
        });

        // placeholders for wiring later
        document.getElementById('email-form')?.addEventListener('submit', (e) => {
            e.preventDefault();
            const box = document.getElementById('auth-status');
            box.textContent = 'Email/password auth will be enabled after backend wiring.';
            box.classList.remove('hidden');
            box.focus();
        });

        document.getElementById('google-btn')?.addEventListener('click', () => {
            const box = document.getElementById('auth-status');
            box.textContent = 'Google sign-in will open after wiring Supabase Auth.';
            box.classList.remove('hidden');
            box.focus();
        });
    </script>
