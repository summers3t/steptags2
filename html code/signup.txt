    <script>
        document.addEventListener('DOMContentLoaded', () => { window.feather && window.feather.replace(); });

        const statusBox = document.getElementById('signup-status');
        const pwd = document.getElementById('password');
        const confirmPwd = document.getElementById('confirm');
        const strengthBar = document.getElementById('pwd-strength');

        // password visibility toggles
        function toggleVisibility(inputId, btnId) {
            const input = document.getElementById(inputId);
            const btn = document.getElementById(btnId);
            btn?.addEventListener('click', () => {
                const visible = input.getAttribute('type') === 'text';
                input.setAttribute('type', visible ? 'password' : 'text');
                btn.setAttribute('aria-label', visible ? 'Show password' : 'Hide password');
                btn.innerHTML = '';
                const i = document.createElement('i');
                i.setAttribute('data-feather', visible ? 'eye' : 'eye-off');
                i.className = 'w-5 h-5';
                btn.appendChild(i);
                window.feather && window.feather.replace();
            });
        }
        toggleVisibility('password', 'toggle-pass');
        toggleVisibility('confirm', 'toggle-confirm');

        // naive strength indicator
        function calcStrength(v) {
            let s = 0;
            if (v.length >= 8) s += 30;
            if (/[A-Z]/.test(v)) s += 20;
            if (/[a-z]/.test(v)) s += 20;
            if (/\d/.test(v)) s += 15;
            if (/[^A-Za-z0-9]/.test(v)) s += 15;
            return Math.min(s, 100);
        }
        pwd.addEventListener('input', () => {
            const w = calcStrength(pwd.value);
            strengthBar.style.width = w + '%';
        });

        // form placeholder
        document.getElementById('signup-form')?.addEventListener('submit', (e) => {
            e.preventDefault();
            if (pwd.value !== confirmPwd.value) {
                statusBox.textContent = 'Passwords do not match.';
                statusBox.classList.remove('hidden');
                statusBox.focus();
                return;
            }
            if (!document.getElementById('terms').checked) {
                statusBox.textContent = 'Please accept the Terms and Privacy Policy.';
                statusBox.classList.remove('hidden');
                statusBox.focus();
                return;
            }
            statusBox.textContent = 'Signup will be enabled after backend wiring.';
            statusBox.classList.remove('hidden');
            statusBox.focus();
        });

        document.getElementById('google-btn')?.addEventListener('click', () => {
            statusBox.textContent = 'Google signup will open after wiring Supabase Auth.';
            statusBox.classList.remove('hidden');
            statusBox.focus();
        });
    </script>
